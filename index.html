local Player = { name = name , score = score , active = isActive }

function Player:new(name, score)
local obj = { name = name , score = score }
setmetatable(obj, self)
self.__index = self
return obj
end

function Player:displayStatus()
if self.active then
print(self.name .. " is active with score " .. self.score)
else
print(self.name .. " is inactive" )
end
end
